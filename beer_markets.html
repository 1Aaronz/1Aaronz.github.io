beer_test <- beer_markets |> 
  filter(rnd <= 0.23)%>%

model1 <- lm(log(price_floz) ~ market + brand + I(container=="CAN") + log(beer_floz), 
             data = beer_train)

model2 <- lm(log(price_floz) ~ market + I(container=="CAN") + brand * log(beer_floz), 
             data = beer_train)

model3 <- lm(log(price_floz) ~ market + I(container=="CAN") + brand * promo * log(beer_floz), 
             data = beer_train)

Requirements:

  • Interpret each coefficient relative to the reference market (BUFFALO-ROCHESTER), holding other variables fixed.
  • Since the outcome is in logs, convert each coefficient to an approximate percent difference using:
    • Approximation: \(100\times \beta\) (when \(|\beta|\) is small)

# Model 3
b3 <- coef(model3)
main <- b3["log(beer_floz)"]
promo_main <- b3["promoTRUE:log(beer_floz)"]

# Get brand names and terms
brands <- c("BUD_LIGHT", "BUSCH_LIGHT", "COORS_LIGHT", "MILLER_LITE", "NATURAL_LIGHT")
brand_terms <- b3[grepl("brand.*:log\\(beer_floz\\)$", names(b3))]
brand_promo_terms <- b3[grepl("brand.*:promo.*:log\\(beer_floz\\)", names(b3))]

promo_false_values <- numeric(5)
promo_true_values <- numeric(5)

promo_false_values[1] <- main
promo_true_values[1] <- main + promo_main

for(i in 2:5) {
  promo_false_values[i] <- main + brand_terms[i-1]
  promo_true_values[i] <- main + brand_terms[i-1] + promo_main + brand_promo_terms[i-1]
}

results <- data.frame(
  brand = brands,
  promo_false = promo_false_values,
  promo_true = promo_true_values
)

results$difference <- results$promo_true - results$promo_false

# Get brand names and terms

results <- data.frame(
